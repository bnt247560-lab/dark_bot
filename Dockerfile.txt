# استخدام نسخة خفيفة من Rust للتحويل
FROM rust:1.70-slim-buster as builder

# تثبيت التبعيات اللازمة للبناء
RUN apt-get update && apt-get install -y pkg-config libssl-dev

WORKDIR /usr/src/dark_bot
COPY . .
RUN cargo build --release

# مرحلة التشغيل (Runtime)
FROM debian:buster-slim

# تثبيت FFmpeg و yt-dlp و Python (لأن yt-dlp يحتاجه)
RUN apt-get update && apt-get install -y \
    ffmpeg \
    python3 \
    python3-pip \
    && pip3 install yt-dlp \
    && rm -rf /var/lib/apt/lists/*

WORKDIR /app
COPY --from=builder /usr/src/dark_bot/target/release/dark_bot .
RUN mkdir downloads

# متغير البيئة لتوكن البوت
ENV TELOXIDE_TOKEN="YOUR_BOT_TOKEN_HERE"

CMD ["./dark_bot"]
